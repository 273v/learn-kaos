import { Card, CardGrid, Badge } from '@astrojs/starlight/components';
import { examples } from '~/data/examples.ts';

Every example below is a self-contained [PEP 723](https://peps.python.org/pep-0723/)
script. Run any of them with **one command** — no clone, no virtualenv:

```bash
uv run https://raw.githubusercontent.com/273v/learn-kaos/main/examples/<id>.py
```

Cards marked <Badge text="offline" variant="success" /> need no API key and run
deterministically in this site's CI. <Badge text="needs key" variant="caution" />
marks examples that call a live LLM provider.

export const groups = [
  { label: 'Golden path & getting started', match: (e) => /^(tutorials|get-started)\//.test(e.tutorial ?? '') },
  { label: 'How-to recipes', match: (e) => (e.tutorial ?? '').startsWith('how-to/') },
  { label: 'Legal & financial use cases', match: (e) => (e.tutorial ?? '') === 'use-cases' },
  // Catch-all so no example is ever dropped from the gallery.
  { label: 'More examples', match: (e) => !/^(tutorials|get-started|how-to)\//.test(e.tutorial ?? '') && (e.tutorial ?? '') !== 'use-cases' },
];

export const renderCard = (e) => (
  <Card key={e.id} title={e.title} icon="rocket">
    {e.summary}

    {e.offline_ok ? <Badge text="offline" variant="success" /> : <Badge text="needs key" variant="caution" />}{' '}
    <Badge text={e.level} variant="default" />

    <p><code>{`uv run examples/${e.id}.py`}</code></p>

    <p>Packages: {e.packages.join(', ')}</p>
  </Card>
);

{groups.map((g) => {
  const items = examples.filter(g.match);
  if (!items.length) return null;
  return (
    <section key={g.label}>
      <h2>{g.label} <span style="font-weight:400;opacity:.6">({items.length})</span></h2>
      <CardGrid>{items.map(renderCard)}</CardGrid>
    </section>
  );
})}

:::note
The gallery grows each milestone — cards are generated from
[`examples/index.toml`](https://github.com/273v/learn-kaos/blob/main/examples/index.toml),
so adding an example is a one-line manifest entry plus the script.
:::
